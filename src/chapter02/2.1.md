ADD-BINARY-INTEGERS(A, B, n)
    let C[0:n]
    carry = 0
    for i = 0 to n - 1
        sum = A[i] + B[i] + carry
        C[i] = sum mod 2
        carry = sum / 2
    C[n] = carry
    return C
